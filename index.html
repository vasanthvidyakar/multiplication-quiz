<script>

// Story-based 5 questions for party prep theme

const questions = [

  {

    story: "Ria is preparing party decorations. She has <b>6 packs of balloons</b>, each with <b>13 balloons</b>. How many balloons in total?",

    img: "🎈",

    multA: 6,

    multB: 13,

    split: [10, 3]

  },

  {

    story: "Next, Ria packs <b>5 boxes of cupcakes</b>. Each box contains <b>14 cupcakes</b>. How many cupcakes altogether?",

    img: "🧁",

    multA: 5,

    multB: 14,

    split: [10, 4]

  },

  {

    story: "Ria sets up <b>4 baskets</b> with <b>18 strawberries</b> each. How many strawberries are there in total?",

    img: "🍓",

    multA: 4,

    multB: 18,

    split: [10, 8]

  },

  {

    story: "At the party area, there are <b>7 ice cream stands</b>, each selling <b>15 cones</b> daily. How many cones are sold altogether?",

    img: "🍦",

    multA: 7,

    multB: 15,

    split: [10, 5]

  },

  {

    story: "Finally, Ria has <b>8 tables</b>. Each table has <b>12 chairs</b>. How many chairs in total?",

    img: "🪑",

    multA: 8,

    multB: 12,

    split: [10, 2]

  }

];


let index = 0, inputs = Array(9).fill('');

let focusId = null;

let firstTryCorrect = Array(questions.length).fill(false);

let solutionViewedTracker = Array(questions.length).fill(false);


function updateProgress() {

  const total = questions.length;

  const done = index;

  document.getElementById('progressHolder').innerText = `Question ${done + 1} of ${total}`;

}


function renderQuestion() {

  const q = questions[index];

  inputs = Array(9).fill('');

  focusId = 0;

  document.getElementById('solution-container').style.display='none';

  document.getElementById('retry-container').style.display='none';

  document.getElementById('result-container').style.display='none';

  document.getElementById('quiz-container').style.display='block';

  updateProgress();


  const markup = `

    <div class="quizimg">${q.img}</div>

    <h2>Party Prep Multiplication</h2>

    <div class="story-row">${q.story}</div>

    <div class="step-row">${q.multA} × ${q.multB} = ${q.multA} × ( <input class="quizbox" id="in0" maxlength="3"> + <input class="quizbox" id="in1" maxlength="3"> )</div>

    <div class="step-row">= <input class="quizbox" id="in2" maxlength="3"> × <input class="quizbox" id="in3" maxlength="3"> + <input class="quizbox" id="in4" maxlength="3"> × <input class="quizbox" id="in5" maxlength="3"></div>

    <div class="step-row">= <input class="quizbox" id="in6" maxlength="4"> + <input class="quizbox" id="in7" maxlength="4"></div>

    <div class="step-row">= <input class="quizbox" id="in8" maxlength="5"></div>

    <div class="num-pad">${[1,2,3,4,5,6,7,8,9,0].map(n=>`<button onclick="insertNum('${n}')">${n}</button>`).join('')}

      <button onclick="deleteNum()">&#9003;</button>

    </div>

    <button class="action-btn" onclick="submitAns()">SUBMIT</button>

  `;

  document.getElementById('quiz-container').innerHTML = markup;

  let allBoxes = document.querySelectorAll('.quizbox');

  allBoxes.forEach((inp, i) => {

    inp.value = '';

    inp.onfocus=()=>{focusId = i;};

    inp.oninput = e => handleType(e,i);

  });

  allBoxes[0].focus();

}


function insertNum(num) {

  if (focusId !== null) {

    let box = document.getElementById(`in${focusId}`);

    box.value += num;

    inputs[focusId] = box.value;

    box.focus();

  }

}

function deleteNum() {

  if (focusId !== null) {

    let box = document.getElementById(`in${focusId}`);

    box.value = box.value.slice(0, -1);

    inputs[focusId] = box.value;

    box.focus();

  }

}

function handleType(e,i) {

  inputs[i] = e.target.value.replace(/[^0-9]/g,'');

}


function submitAns() {

  const q = questions[index], [A,B] = [q.multA, q.multB], [S1, S2] = q.split;

  const correctAnswers = [

    S1, S2,

    A, S1,

    A, S2,

    A*S1, A*S2,

    A*S1 + A*S2

  ];

  let ok = true;

  for(let i=0; i<9; i++) {

    if(!inputs[i] || inputs[i] === "" || inputs[i] != correctAnswers[i]) {

      ok = false;

      break;

    }

  }

  if(ok && !firstTryCorrect[index]) firstTryCorrect[index] = true;

  if(ok) {

    showSolution(true);

  } else {

    encourageRetry();

  }

}


function encourageRetry() {

  document.getElementById('quiz-container').style.display='none';

  const cheers = [

    "Almost there! Give it another go! 🌈",

    "Oops! Check your math magic and try again! 🪄",

    "You can do it! Think about each step. 💡"

  ];

  const cheer = cheers[Math.floor(Math.random()*cheers.length)];

  document.getElementById('retry-container').innerHTML = `

    <div class="feedback" style="color:#f2395a;">${cheer}</div>

    <button class="action-btn" onclick="tryAgain()">Try Again</button>

    <button class="action-btn secondary" onclick="showSolution(false)">See Solution</button>

  `;

  document.getElementById('retry-container').style.display='block';

}


function tryAgain() {

  document.getElementById('retry-container').style.display='none';

  document.getElementById('quiz-container').style.display='block';

}


function showSolution(correct) {

  if(!correct) solutionViewedTracker[index]=true;

  document.getElementById('quiz-container').style.display='none';

  document.getElementById('retry-container').style.display='none';

  const q=questions[index], [A,B]=[q.multA,q.multB], [S1,S2]=q.split;

  const products=[A*S1, A*S2], total=products[0]+products[1];

  const markup=`

    <div class="quizimg">${q.img}</div>

    <h2>Great Job!</h2>

    <div class="story-row" style="color:#178a66;">${q.story}</div>

    <table class="solution-table" aria-label="Solution steps">

      <tr><td>${A} × ${B}</td><td>= ${A} × ( ${S1} + ${S2} )</td></tr>

      <tr><td></td><td>= ${A} × ${S1} + ${A} × ${S2}</td></tr>

      <tr><td></td><td>= ${products[0]} + ${products[1]}</td></tr>

      <tr><td></td><td>= <b style="color:#e6502c; font-size:1.12em;">${total}</b></td></tr>

    </table>

    <div class="feedback" style="color:#45ad35; font-weight:bold;">

      ${correct ? '🎈 Hooray! You solved it!' : 'Keep going, you will get better every time!'}

    </div>

    <button class="action-btn" onclick="${index < questions.length-1 ? 'nextQ()' : 'showResults()'}">

      ${index < questions.length-1 ? 'Next Question ➡️' : 'Finish Quiz 🏁'}

    </button>

  `;

  document.getElementById('solution-container').innerHTML=markup;

  document.getElementById('solution-container').style.display='block';

}


function nextQ() {

  index++;

  if(index >= questions.length) {

    showResults();

  } else {

    renderQuestion();

  }

}


function showResults() {

  document.getElementById('quiz-container').style.display='none';

  document.getElementById('retry-container').style.display='none';

  document.getElementById('solution-container').style.display='none';

  document.getElementById('result-container').style.display='block';

  document.getElementById('progressHolder').innerText = '';


  const total = questions.length;

  const correctCount = firstTryCorrect.reduce((acc, v) => acc + (v ? 1 : 0), 0);

  const viewedCount = solutionViewedTracker.reduce((acc, v) => acc + (v ? 1 : 0), 0);


  let stars = 5;

  if(correctCount < total) stars--;

  stars -= viewedCount;

  if(stars < 1) stars = 1;


  let starMarkup = '';

  for(let i=0;i<stars;i++) starMarkup += '⭐';

  for(let i=stars;i<5;i++) starMarkup += '☆';


  const markup = `

    <h2>Quiz Results</h2>

    <div style="font-size: 18px; margin-top: 20px;">

      You answered <b>${correctCount} out of ${total}</b> questions correctly on your first attempt.<br>

      You viewed the solution <b>${viewedCount} time${viewedCount !== 1 ? 's' : ''}</b>.<br><br>

      Your star rating:<br>

      <div class="stars">${starMarkup}</div>

    </div>

    <button class="action-btn" onclick="restartQuiz()">Restart Quiz 🔄</button>

  `;

  document.getElementById('result-container').innerHTML = markup;

}


function restartQuiz() {

  index = 0;

  firstTryCorrect.fill(false);

  solutionViewedTracker.fill(false);

  document.getElementById('result-container').style.display = 'none';

  document.getElementById('progressHolder').innerText = `Question 1 of ${questions.length}`;

  renderQuestion();

}


renderQuestion();

</script>
